OpenEuler22.03 SP2 LTS安装Minio2023集群

发表于 2023-12-21 分类于 操作系统 ， 软件安装 本文字数： 2.1k 阅读时长 ≈ 8 分钟

1.环境规划

序号 IP Hostname CPU/内存配置 系统盘 数据盘
1 192.168.12.210 minio-nginx 4C+16G 80G 200G
2 192.168.12.201 minio-data1 8C+16G 80G 5000G
3 192.168.12.202 minio-data2 8C+16G 80G 5000G
4 192.168.12.203 minio-data3 8C+16G 80G 5000G
5 192.168.12.204 minio-data4 8C+16G 80G 5000G

安装系统时，root默认密码为“~!@`12qwe”

内部DNS服务器：http://minio.yicr.dap ==》192.168.12.210

操作系统：OpenEuler22.03 SP2 LTS

Nginx版本：nginx-1.24.0

Minio版本：RELEASE.2023-12-20T01-00-02Z

2.Minio-nginx

2.1更新系统并设置selinux

系统安装完毕后，先行更新系统，确保系统组件为最新状态。

#更新系统
dnf upgrade -y

# 调整selinux策略为宽容模式(nginx自启动的selinux配置复杂)
vi /etc/selinux/config
# 设置 SELINUX=permissive

2.2通过LV方式挂载数据盘

#查看磁盘情况
fdisk -l
#给“/dev/vdb分区”并设置分区类型为Linux LVM。
fdisk /dev/vdb

分区输入：

Command (m for help): n
Partition type
   p   primary (0 primary, 0 extended, 4 free)
   e   extended (container for logical partitions)
Select (default p): p
Partition number (1-4, default 1): 回车
First sector (2048-419430399, default 2048): 回车
Last sector, +/-sectors or +/-size{K,M,G,T,P} (2048-419430399, default 419430399): 回车

Created a new partition 1 of type 'Linux' and of size 200 GiB.

Command (m for help): t
Selected partition 1
Hex code or alias (type L to list all): 8e
Changed type of partition 'Linux' to 'Linux LVM'.

Command (m for help): w
The partition table has been altered.
Calling ioctl() to re-read partition table.
Syncing disks.

继续执行：

#再次查看分区情况
fdisk -l
#显示结果已创建vdb1分区
Device     Boot Start       End   Sectors  Size Id Type
/dev/vdb1        2048 419430399 419428352  200G 8e Linux LVM
#格式化分区（也可等到LV格式化）
mkfs.ext4 /dev/vdb1

把分区添加到物理卷(PV)

# 为`/dev/vdb1` 分区创建PV：
pvcreate /dev/vdb1

创建新的卷组(VG)

# 如果VG还不存在，则先创建VG，否则可以将PV加到现有的VG中来扩容VG。
# 创建名为vg_minio_data 的VG，包含/dev/vdb1 PV。
vgcreate vg_minio_data /dev/vdb1 
# 可运行vgscan来列出VG
# 运行vgdisplay vg_dlp查看VG详细信息查看VG详细信息

创建逻辑卷（LV）

# 从vg_minio_data VG中创建一个名为lv_minio_data的LV，大小为180G。
lvcreate -L 180G -n lv_minio_data vg_minio_data
# 可运行lvs 查看LV报告或运行lvscan查看LV信息
# 运行lvdisplay /dev/vg_minio_data/lv_minio_data 可查看LV的详细信息。

将LV挂载到目录

# 创建待挂载的目录
mkdir /app
# 挂载前先格式化LV
mkfs.ext4 /dev/vg_minio_data/lv_minio_data
# 挂载LV
mount /dev/vg_minio_data/lv_minio_data /app

永久挂载LV到目录

# 编辑/etc/fstab文件
vi /etc/fstab
# 添加以下行后保存退出
/dev/vg_minio_data/lv_minio_data        /app         ext4    defaults        0 0

2.3下载并编译安装Nginx

# 安装必要依赖
dnf -y install tar gcc make pcre pcre-devel zlib zlib-devel openssl openssl-devel
# 创建nginx安装文件夹
mkdir /app/nginx
# 下载nginx源码包 （下载到了/app下）
wget https://nginx.org/download/nginx-1.24.0.tar.gz
# 解压源码包
tar -zxvf nginx-1.24.0.tar.gz 
# 安装nginx
cd /app/nginux-1.24.0;

# 配置nginx安装路径及安装模块
./configure --prefix=/app/nginx    --with-http_stub_status_module --with-http_ssl_module --with-http_v2_module --with-http_sub_module --with-http_gzip_static_module --with-pcre --with-http_realip_module
#--prefix 指定安装路径
#--with-http_stub_status_module    允许查看nginx状态的模块
# --with-http_ssl_module           支持https的模块

# 编译&&安装
make && make install

修改nginx运行用户为root：

vi /app/nginx/config/nginx.conf
# 移除user nobody 前的“#”，修改noby为root

设置nginx开机自启：

# 创建启动脚本
vi /etc/systemd/system/nginx.service

nginx.service内容：

[Unit]
Description=nginx
After=network.target

[Service]
Type=forking
ExecStart=/app/nginx/sbin/nginx -c /app/nginx/conf/nginx.conf
ExecReload=/app/nginx/sbin/nginx -s reload
ExecStop=/app/nginx/sbin/nginx -s quit
PrivateTmp=true

[Install]
WantedBy=multi-user.target

# 重载配置
systemctl daemon-reload
# 加入自启
systemctl enable nginx
# 启动nginx
systemctl start nginx

2.4防火墙配置

# 配置防火墙放行5432端口
sudo firewall-cmd --add-port=80/tcp --permanent  
sudo firewall-cmd --add-port=443/tcp --permanent
# 重新加载防火墙配置文件
sudo firewall-cmd --reload

3.Minio-data

包含minio-data1/2/3/4四台服务器，配置相同。（hostname请按照文章开头设置）

3.1挂载大容量硬盘（大于2T）

# 待分区：/dev/vdb 4.88T
# 硬盘容量超过2T，需要使用parted进行gpt分区
[root@openeuler-base ~]# parted /dev/vdb 
GNU Parted 3.5
Using /dev/vdb
Welcome to GNU Parted! Type 'help' to view a list of commands.
(parted) mklabel gpt                  # 转为gpt分区                                               
(parted) mkpart primary xfs 0% 100%  # 创建xfs文件格式的主分区                                          
(parted) p                    # print打印                                            
Model: Virtio Block Device (virtblk)
Disk /dev/vdb: 5369GB
Sector size (logical/physical): 512B/512B
Partition Table: gpt
Disk Flags: 

Number  Start   End     Size    File system  Name     Flags
 1      17.4kB  5369GB  5369GB  ext4         primary

(parted) toggle 1 lvm          #硬盘类型更改为LVM                                           
(parted) p                                                                
Model: Virtio Block Device (virtblk)
Disk /dev/vdb: 5369GB
Sector size (logical/physical): 512B/512B
Partition Table: gpt
Disk Flags: 

Number  Start   End     Size    File system  Name     Flags
 1      17.4kB  5369GB  5369GB  ext4         primary  lvm

(parted) quit                 # 退出 parted 操作                                            
Information: You may need to update /etc/fstab.

[root@openeuler-base ~]# partprobe     #系统重读分区表

创建PV（物理卷）、VG（卷组）、LV（逻辑卷）：

# 把分区加到物理卷（创建PV）
# 运行pvscan命令列出所有的PV
# 运行pvdisplay /dev/vdb1 查看PV详细信息
pvcreate /dev/vdb1 

# 创建新的卷组（VG）
# 运行vgscan来列出VG
# 运行vg_minio_data查看VG详细信息
vgcreate vg_minio_data /dev/vdb1

# 创建逻辑卷（LV）
# 运行lvscan查看LV信息
# 运行lvdisplay /dev/vg_minio_data/lv_minio_data可查看LV的详细信息
# 自vg_minio_data中创建一个名为lv_minio_data的LV，大小为4.8T
lvcreate -L 4.8T -n lv_minio_data vg_minio_data

# 格式化LV为xfs文件格式
mkfs -t xfs /dev/vg_minio_data/lv_minio_data

# 在“/app”目录下创建/minio/run、/minio/data和/minio/config文件夹，分别用于存储执行脚本、数据文件和配置文件：
mkdir -p /app/minio/run /app/minio/data /app/minio/config
# 临时挂载LV至挂载目录/app/minio/data
mount /dev/vg_minio_data/lv_minio_data /app/minio/data
# 永久挂载LV
vi /etc/fstab
# 文件末尾添加如下行后保存退出
/dev/vg_minio_data/lv_minio_data /app/minio/data              xfs    defaults        0 0

3.2系统调优

修改系统最大文件连接数：

vi /etc/security/limits.conf

# 在文件末尾添加如下两行：
*               soft    nofile          65535
*               hard    nofile          65535
# 保存退出

# 重新加载系统配置
sysctl -p
#重启生效
reboot
# 可通过一下命令查看当前系统连接数
ulimit -n   #同一时间最多可开启的文件数
ulimit -a   #显示当前资源限制

修改host（vi /etc/hosts），添加如下内容：

192.168.12.201 minio-data1
192.168.12.202 minio-data2
192.168.12.203 minio-data3
192.168.12.204 minio-data4

3.3防火墙配置：

sudo firewall-cmd --add-port=9000/tcp --permanent  
sudo firewall-cmd --add-port=9001/tcp --permanent
sudo firewall-cmd --reload

3.4minio安装配置

下载minio到“/app/minio/run”目录：

cd /app/minio/run
wget https://dl.min.io/server/minio/release/linux-amd64/minio

在“/app/minio/run”创建minio集群启动脚本run.sh

vi /app/minio/run/run.sh

minio集群启动脚本run.sh内容：

# run.sh文件内容
#!/bin/bash
export MINIO_ROOT_USER=DapMinio
export MINIO_ROOT_PASSWORD=DapYicr2023!.!

export MINIO_SERVER_URL=http://minio.yicr.dap
export MINIO_BROWSER_REDIRECT_URL=http://minio.yicr.dap/ui

/app/minio/run/minio server --config-dir /app/minio/config --address ":9000" --console-address ":9001" \
http://minio-data1:9000/app/minio/data http://minio-data2:9000/app/minio/data http://minio-data3:9000/app/minio/data http://minio-data4:9000/app/minio/data > /app/minio/log/minio_server.log

给minio和启动脚本授予执行权限：

chmod +x /app/minio/run/minio
chmod +x /app/minio/run/run.sh

配置为系统服务

vi /usr/lib/systemd/system/minio.service

文件内容：

[Unit]
Description=Minio service
Documentation=https://docs.minio.io/
 
[Service]
WorkingDirectory=/app/minio/run/
ExecStart=/app/minio/run/run.sh
 
Restart=on-failure
RestartSec=5
 
[Install]
WantedBy=multi-user.target

4.Nginx负载均衡配置

修改“minio-nginx”主机的Nginx.conf配置：

user  root;
worker_processes  1;

#error_log  logs/error.log;
#error_log  logs/error.log  notice;
#error_log  logs/error.log  info;

#pid        logs/nginx.pid;


events {
    worker_connections  1024;
}


http {
    include       mime.types;
    default_type  application/octet-stream;

    #log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
    #                  '$status $body_bytes_sent "$http_referer" '
    #                  '"$http_user_agent" "$http_x_forwarded_for"';

    #access_log  logs/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    #keepalive_timeout  0;
    keepalive_timeout  65;

    #gzip  on;

	upstream minio_api {
        least_conn;
        server minio-data1:9000;
        server minio-data2:9000;
        server minio-data3:9000;
        server minio-data4:9000;
	}

	upstream minio_console {
        least_conn;
        server minio-data1:9001;
        server minio-data2:9001;
        server minio-data3:9001;
        server minio-data4:9001;
	}
    server {
    listen       80;
    listen  [::]:80;
    server_name  minio.yicr.dap;

    # Allow special characters in headers
    ignore_invalid_headers off;
    # Allow any size file to be uploaded.
    # Set to a value such as 1000m; to restrict file size to a specific value
    client_max_body_size 0;
    # Disable buffering
    proxy_buffering off;
    proxy_request_buffering off;

    location / {
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_connect_timeout 300;
        # Default is HTTP/1, keepalive is only enabled in HTTP/1.1
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        chunked_transfer_encoding off;

        proxy_pass http://minio_api; # This uses the upstream directive definition to load balance
        }

    location /ui/ {
        rewrite ^/ui/(.*) /$1 break;
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-NginX-Proxy true;

        # This is necessary to pass the correct IP to be hashed
        real_ip_header X-Real-IP;

        proxy_connect_timeout 300;

        # To support websockets in MinIO versions released after January 2023
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        # Some environments may encounter CORS errors (Kubernetes + Nginx Ingress)
        # Uncomment the following line to set the Origin request to an empty string
        # proxy_set_header Origin '';

        chunked_transfer_encoding off;

        proxy_pass http://minio_console; # This uses the upstream directive definition to load balance
        }
    }

}

DNS配置成功后可以直接通过http://minio.yicr.dap/ui 来访问控制台。
